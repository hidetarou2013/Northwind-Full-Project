232    @GetMapping(produces = Datatables.MEDIA_TYPE, name = "datatables", value = "/dt")
233    @ResponseBody
234    public ResponseEntity<ConvertedDatatablesData<Category>> datatables(DatatablesColumns datatablesColumns, GlobalSearch search, DatatablesPageable pageable, @RequestParam("draw") Integer draw) {
235        Page<Category> categories = getCategoryService().findAll(search, pageable);
236        long totalCategoriesCount = categories.getTotalElements();
237        if (search != null && StringUtils.isNotBlank(search.getText())) {
238            totalCategoriesCount = getCategoryService().count();
239        }
240        ConvertedDatatablesData<Category> datatablesData = new ConvertedDatatablesData<Category>(categories, totalCategoriesCount, draw, getConversionService(), datatablesColumns);
241        return ResponseEntity.ok(datatablesData);
242    }

361    /**
362     * Method that obtains the filtered and ordered records using the Datatables information and
363     * export them to a new report file. (It ignores the current pagination).
364     *
365     * To generate the report file it uses the `DynamicJasper` library
366     * (http://dynamicjasper.com). This library allows developers to generate reports dynamically
367     * without use an specific template to each entity.
368     *
369     * To customize the appearance of ALL generated reports, you could customize the
370     * "export_default.jrxml" template located in "src/main/resources/templates/reports/". However,
371     * if you want to customize the appearance of this specific report, you could create a new
372     * ".jrxml" file and provide it to the library replacing the `builder.setTemplateFile();`
373     * operation used in this implementation.
374     *
375     * @param search GlobalSearch that contains the filter provided by the Datatables component.
376     * @param pageable Pageable that contains the Sort info provided by the Datatabes component.
377     * @param datatablesColumns Columns displayed in the Datatables component.
378     * @param response The HttpServletResponse.
379     * @param exporter An specific JasperReportsExporter to be used during export process.
380     * @param fileName The final filename to use.
381     * @param locale The current Locale in the view context.
382     */
383    public void export(GlobalSearch search, @PageableDefault(size = 2147483647) Pageable pageable, String[] datatablesColumns, HttpServletResponse response, JasperReportsExporter exporter, String fileName, Locale locale) {
384        // Obtain the filtered and ordered elements
385        Page<Category> categories = getCategoryService().findAll(search, pageable);
386        // Prevent generation of reports with empty data
387        if (categories == null || categories.getContent().isEmpty()) {
388            return;
389        }
390        // Creates a new ReportBuilder using DynamicJasper library
391        FastReportBuilder builder = new FastReportBuilder();
392        // IMPORTANT: By default, this application uses "export_default.jrxml"
393        // to generate all reports. If you want to customize this specific report,
394        // create a new ".jrxml" template and customize it. (Take in account the
395        // DynamicJasper restrictions:
396        // http://dynamicjasper.com/2010/10/06/how-to-use-custom-jrxml-templates/)
397        builder.setTemplateFile("templates/reports/export_default.jrxml");
398        // The generated report will display the same columns as the Datatables component.
399        // However, this is not mandatory. You could edit this code if you want to ignore
400        // the provided datatablesColumns
401        if (datatablesColumns != null) {
402            for (String column : datatablesColumns) {
403                // Delegates in addColumnToReportBuilder to include each datatables column
404                // to the report builder
405                addColumnToReportBuilder(column, builder, locale, fileName);
406            }
407        }
408        // This property resizes the columns to use full width page.
409        // Set false value if you want to use the specific width of each column.
410        builder.setUseFullPageWidth(true);
411        // Creates a new Jasper Reports Datasource using the obtained elements
412        JRDataSource ds = new JRBeanCollectionDataSource(categories.getContent());
413        // Generates the JasperReport
414        JasperPrint jp;
415        try {
416            jp = DynamicJasperHelper.generateJasperPrint(builder.build(), new ClassicLayoutManager(), ds);
417        } catch (JRException e) {
418            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
419            throw new ExportingErrorException(errorMessage);
420        }
421        // Converts the JaspertReport element to a ByteArrayOutputStream and
422        // write it into the response stream using the provided JasperReportExporter
423        try {
424            exporter.export(jp, fileName, response);
425        } catch (JRException e) {
426            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
427            throw new ExportingErrorException(errorMessage);
428        } catch (IOException e) {
429            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
430            throw new ExportingErrorException(errorMessage);
431        }
432    }

485    /**
486     * This method contains all the entity fields that are able to be displayed in a
487     * report. The developer could add a new column to the report builder providing the
488     * field name and the builder where the new field will be added as column.
489     *
490     * @param columnName the field name to show as column
491     * @param builder The builder where the new field will be added as column.
492     */
493    public void addColumnToReportBuilder(String columnName, FastReportBuilder builder, Locale locale, String fileName) {
494        try {
495            if (columnName.equals("id")) {
496                builder.addColumn(getMessageSource().getMessage("label_category_id", null, "Id", locale), "id", Long.class.getName(), 50);
497            } else if (columnName.equals("version")) {
498                builder.addColumn(getMessageSource().getMessage("label_category_version", null, "Version", locale), "version", Long.class.getName(), 100);
499            } else if (columnName.equals("name")) {
500                builder.addColumn(getMessageSource().getMessage("label_category_name", null, "Name", locale), "name", String.class.getName(), 100);
501            } else if (columnName.equals("description")) {
502                builder.addColumn(getMessageSource().getMessage("label_category_description", null, "Description", locale), "description", String.class.getName(), 100);
503            } else if (columnName.equals("createdDate")) {
504                builder.addColumn(getMessageSource().getMessage("label_category_createddate", null, "Created Date", locale), "createdDate", Calendar.class.getName(), 100);
505            } else if (columnName.equals("createdBy")) {
506                builder.addColumn(getMessageSource().getMessage("label_category_createdby", null, "Created By", locale), "createdBy", String.class.getName(), 100);
507            } else if (columnName.equals("modifiedDate")) {
508                builder.addColumn(getMessageSource().getMessage("label_category_modifieddate", null, "Modified Date", locale), "modifiedDate", Calendar.class.getName(), 100);
509            } else if (columnName.equals("modifiedBy")) {
510                builder.addColumn(getMessageSource().getMessage("label_category_modifiedby", null, "Modified By", locale), "modifiedBy", String.class.getName(), 100);
511            }
512        } catch (ColumnBuilderException e) {
513            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
514            throw new ExportingErrorException(errorMessage);
515        } catch (ClassNotFoundException e) {
516            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
517            throw new ExportingErrorException(errorMessage);
518        }
519    }
