001package org.springframework.roo.northwind.web.advices;
002
003import io.springlets.web.mvc.util.concurrency.ConcurrencyException;
004
005import org.springframework.web.bind.annotation.ControllerAdvice;
006import org.springframework.web.bind.annotation.ExceptionHandler;
007import org.springframework.web.servlet.ModelAndView;
008
009/**
010 * = ConcurrencyExceptionManager
011 * <p>
012 * Controller Advice that monitors all the existing @Controller's and catch
013 * all the {@link ConcurrencyException} that could appear.
014 */
015@ControllerAdvice
016public class ConcurrencyExceptionManager {
017
018    /**
019     * ExceptionHandler that manages the {@link ConcurrencyException}. Using the information
020     * available inside the {@link ConcurrencyException} object, it delegates in the manager
021     * that throws the exception to populate the concurrency form and to return a valid view.
022     * <p>
023     * The controller is the only one who knows how to manage the concurrency. Also, it allow us to
024     * customize the default behavior of the concurrency exception feature.
025     *
026     * @param ex
027     * @return
028     */
029    @ExceptionHandler(ConcurrencyException.class)
030    public ModelAndView concurrencyException(final ConcurrencyException ex) {
031        return ex.populateAndGetFormView();
032    }
033
034
035}
