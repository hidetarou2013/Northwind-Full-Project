356    /**
357     * TODO Auto-generated method documentation
358     *
359     * @param country
360     * @param customerOrders
361     * @return Country
362     */
363    @Transactional
364    public Country setCustomerOrders(Country country, Iterable<Long> customerOrders) {
365        List<CustomerOrder> items = getCustomerOrderService().findAll(customerOrders);
366        Set<CustomerOrder> currents = country.getCustomerOrders();
367        Set<CustomerOrder> toRemove = new HashSet<CustomerOrder>();
368        for (Iterator<CustomerOrder> iterator = currents.iterator(); iterator.hasNext(); ) {
369            CustomerOrder nextCustomerOrder = iterator.next();
370            if (items.contains(nextCustomerOrder)) {
371                items.remove(nextCustomerOrder);
372            } else {
373                toRemove.add(nextCustomerOrder);
374            }
375        }
376        country.removeFromCustomerOrders(toRemove);
377        country.addToCustomerOrders(items);
378        // Force the version update of the parent side to know that the parent has changed
379        // because it has new books assigned
380        country.setVersion(country.getVersion() + 1);
381        return getCountryRepository().save(country);
382    }

496    /**
497     * TODO Auto-generated method documentation
498     *
499     * @param country
500     */
501    @Transactional
502    public void delete(Country country) {
503        // Clear bidirectional one-to-many parent relationship with CustomerOrder
504        for (CustomerOrder item : country.getCustomerOrders()) {
505            item.setCountry(null);
506        }
507        // Clear bidirectional one-to-many parent relationship with Party
508        for (Party item : country.getParties()) {
509            item.setCountry(null);
510        }
511        // Clear bidirectional one-to-many parent relationship with Region
512        for (Region item : country.getRegions()) {
513            item.setCountry(null);
514        }
515        // Clear bidirectional one-to-many parent relationship with Store
516        for (Store item : country.getStores()) {
517            item.setCountry(null);
518        }
519        // Clear bidirectional one-to-many parent relationship with Supplier
520        for (Supplier item : country.getSuppliers()) {
521            item.setCountry(null);
522        }
523        getCountryRepository().delete(country);
524    }
