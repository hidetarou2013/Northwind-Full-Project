081    public void delete(Party party) {
082        // Clear bidirectional many-to-one child relationship with Country
083        if (party.getCountry() != null) {
084            party.getCountry().getParties().remove(party);
085        }
086        // Clear bidirectional many-to-one child relationship with City
087        if (party.getCity() != null) {
088            party.getCity().getParties().remove(party);
089        }
090        // Clear bidirectional many-to-one child relationship with Region
091        if (party.getRegion() != null) {
092            party.getRegion().getParties().remove(party);
093        }
094        getPartyRepository().delete(party);
095    }
