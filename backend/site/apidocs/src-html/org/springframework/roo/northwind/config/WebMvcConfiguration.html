001package org.springframework.roo.northwind.config;
002import org.springframework.roo.addon.web.mvc.controller.annotations.config.RooWebMvcConfiguration;
003import org.springframework.context.ApplicationContextAware;
004import org.springframework.roo.addon.web.mvc.thymeleaf.annotations.RooWebMvcThymeleafUIConfiguration;
005import io.tracee.binding.springmvc.TraceeInterceptor;
006import java.lang.Override;
007import java.util.Locale;
008import org.springframework.beans.factory.annotation.Autowired;
009import org.springframework.boot.autoconfigure.thymeleaf.ThymeleafProperties;
010import org.springframework.context.ApplicationContext;
011import org.springframework.context.annotation.Bean;
012import org.springframework.context.annotation.Configuration;
013import org.springframework.context.annotation.Primary;
014import org.springframework.validation.beanvalidation.LocalValidatorFactoryBean;
015import org.springframework.web.servlet.LocaleResolver;
016import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
017import org.springframework.web.servlet.config.annotation.WebMvcConfigurerAdapter;
018import org.springframework.web.servlet.i18n.LocaleChangeInterceptor;
019import org.springframework.web.servlet.i18n.SessionLocaleResolver;
020import org.thymeleaf.TemplateEngine;
021import org.thymeleaf.spring4.templateresolver.SpringResourceTemplateResolver;
022import org.thymeleaf.spring4.view.ThymeleafViewResolver;
023import org.thymeleaf.templatemode.TemplateMode;
024
025/**
026 * = WebMvcConfiguration
027 TODO Auto-generated class documentation
028 *
029 */
030@RooWebMvcConfiguration(defaultLanguage = "en")
031@RooWebMvcThymeleafUIConfiguration
032@Configuration
033public class WebMvcConfiguration extends WebMvcConfigurerAdapter implements ApplicationContextAware {
034
035    /**
036     * TODO Auto-generated attribute documentation
037     *
038     */
039    @Autowired
040    private ThymeleafProperties thymeleafProperties;
041
042    /**
043     * TODO Auto-generated attribute documentation
044     *
045     */
046    private ApplicationContext applicationContext;
047
048    /**
049     * TODO Auto-generated attribute documentation
050     *
051     */
052    @Autowired
053    private TemplateEngine templateEngine;
054
055    /**
056     * TODO Auto-generated method documentation
057     *
058     * @return LocalValidatorFactoryBean
059     */
060    @Primary
061    @Bean
062    public LocalValidatorFactoryBean validator() {
063        return new LocalValidatorFactoryBean();
064    }
065
066    /**
067     * TODO Auto-generated method documentation
068     *
069     * @return LocaleResolver
070     */
071    @Bean
072    public LocaleResolver localeResolver() {
073        SessionLocaleResolver localeResolver = new SessionLocaleResolver();
074        localeResolver.setDefaultLocale(new Locale("en"));
075        return localeResolver;
076    }
077
078    /**
079     * TODO Auto-generated method documentation
080     *
081     * @return LocaleChangeInterceptor
082     */
083    @Bean
084    public LocaleChangeInterceptor localeChangeInterceptor() {
085        LocaleChangeInterceptor localeChangeInterceptor = new LocaleChangeInterceptor();
086        localeChangeInterceptor.setParamName("lang");
087        return localeChangeInterceptor;
088    }
089
090    /**
091     * TODO Auto-generated method documentation
092     *
093     * @param registry
094     */
095    @Override
096    public void addInterceptors(InterceptorRegistry registry) {
097        registry.addInterceptor(localeChangeInterceptor());
098        registry.addInterceptor(new TraceeInterceptor());
099    }
100
101    /**
102     * TODO Auto-generated method documentation
103     *
104     * @return ThymeleafProperties
105     */
106    public ThymeleafProperties getThymeleafProperties() {
107        return thymeleafProperties;
108    }
109
110    /**
111     * TODO Auto-generated method documentation
112     *
113     * @return TemplateEngine
114     */
115    public TemplateEngine getTemplateEngine() {
116        return templateEngine;
117    }
118
119    /**
120     * TODO Auto-generated method documentation
121     *
122     * @return ApplicationContext
123     */
124    public ApplicationContext getApplicationContext() {
125        return applicationContext;
126    }
127
128    /**
129     * TODO Auto-generated method documentation
130     *
131     * @param applicationContext
132     */
133    public void setApplicationContext(ApplicationContext applicationContext) {
134        this.applicationContext = applicationContext;
135    }
136
137    /**
138     * TODO Auto-generated method documentation
139     *
140     * @return ThymeleafViewResolver
141     */
142    @Bean
143    public ThymeleafViewResolver javascriptThymeleafViewResolver() {
144        ThymeleafViewResolver resolver = new ThymeleafViewResolver();
145        resolver.setTemplateEngine(getTemplateEngine());
146        resolver.setCharacterEncoding("UTF-8");
147        resolver.setContentType("application/javascript");
148        resolver.setViewNames(new String[] { "*.js" });
149        resolver.setCache(getThymeleafProperties().isCache());
150        return resolver;
151    }
152
153    /**
154     * TODO Auto-generated method documentation
155     *
156     * @return SpringResourceTemplateResolver
157     */
158    @Bean
159    public SpringResourceTemplateResolver javascriptTemplateResolver() {
160        SpringResourceTemplateResolver resolver = new SpringResourceTemplateResolver();
161        resolver.setApplicationContext(getApplicationContext());
162        resolver.setPrefix("classpath:/templates/fragments/js/");
163        resolver.setTemplateMode(TemplateMode.JAVASCRIPT);
164        resolver.setCharacterEncoding("UTF-8");
165        resolver.setCheckExistence(true);
166        resolver.setCacheable(getThymeleafProperties().isCache());
167        return resolver;
168    }
169}
