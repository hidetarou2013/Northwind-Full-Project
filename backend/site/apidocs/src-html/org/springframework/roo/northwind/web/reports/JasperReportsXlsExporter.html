001package org.springframework.roo.northwind.web.reports;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005
006import javax.servlet.ServletOutputStream;
007import javax.servlet.http.HttpServletResponse;
008
009import net.sf.jasperreports.engine.JRException;
010import net.sf.jasperreports.engine.JRExporterParameter;
011import net.sf.jasperreports.engine.JasperPrint;
012import net.sf.jasperreports.engine.export.JRXlsAbstractExporterParameter;
013import net.sf.jasperreports.engine.export.JRXlsExporter;
014
015/**
016 * = JasperReportsXlsExporter
017 * 
018 * A JasperReports exporter to export the report in XLS formats. This class
019 * implements the interface {@link JasperReportsExporter}
020 */
021public class JasperReportsXlsExporter implements JasperReportsExporter {
022
023        /**
024         * Generates a ByteArrayOutputStream from the provided JasperReport using
025         * the {@link JRXlsExporter}. After that, the generated bytes array is
026         * written in the {@link HttpServletResponse}
027         * 
028         * @param jp
029         *            The generated JasperReport.
030         * @param fileName
031         *            The fileName of the exported JasperReport
032         * @param response
033         *            The HttpServletResponse where generated report has been
034         *            written
035         * @throws JRException
036         *             during JasperReport export.
037         * @throws IOException
038         *             when writes the ByteArrayOutputStream into the
039         *             HttpServletResponse
040         */
041        @Override
042        public void export(JasperPrint jp, String fileName, HttpServletResponse response) throws JRException, IOException {
043
044                ByteArrayOutputStream baos = new ByteArrayOutputStream();
045
046                // Create a JRXlsExporter instance
047                JRXlsExporter exporter = new JRXlsExporter();
048
049                // Here we assign the parameters jp and baos to the exporter
050                exporter.setParameter(JRExporterParameter.JASPER_PRINT, jp);
051                exporter.setParameter(JRExporterParameter.OUTPUT_STREAM, baos);
052
053                // Excel specific parameters
054                exporter.setParameter(JRXlsAbstractExporterParameter.IS_ONE_PAGE_PER_SHEET, Boolean.FALSE);
055                exporter.setParameter(JRXlsAbstractExporterParameter.IS_REMOVE_EMPTY_SPACE_BETWEEN_ROWS, Boolean.TRUE);
056                exporter.setParameter(JRXlsAbstractExporterParameter.IS_WHITE_PAGE_BACKGROUND, Boolean.FALSE);
057
058                // Retrieve the exported report in PDF format
059                exporter.exportReport();
060
061                // Specifies the response header
062                response.setHeader("Content-Disposition", "inline; filename=" + fileName);
063
064                // Make sure to set the correct content type
065                // Each format has its own content type
066                response.setContentType("application/vnd.ms-excel");
067                response.setContentLength(baos.size());
068
069                // Retrieve the output stream
070                ServletOutputStream outputStream = response.getOutputStream();
071                // Write to the output stream
072                baos.writeTo(outputStream);
073                // Flush the stream
074                outputStream.flush();
075
076        }
077}
