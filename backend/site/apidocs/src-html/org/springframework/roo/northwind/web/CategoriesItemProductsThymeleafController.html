200     * @param locale
201     * @param method
202     * @return Category
203     */
204    @ModelAttribute
205    public Category getCategory(@PathVariable("category") Long id, Locale locale, HttpMethod method) {
206        Category category = null;
207        if (HttpMethod.PUT.equals(method)) {
208            category = categoryService.findOneForUpdate(id);
209        } else {
210            category = categoryService.findOne(id);
211        }
212        if (category == null) {
213            String message = messageSource.getMessage("error_NotFound", new Object[] { "Category", id }, "The record couldn't be found", locale);
214            throw new NotFoundException(message);
215        }
216        return category;
217    }

317    /**
318     * TODO Auto-generated method documentation
319     *
320     * @param category
321     * @param products
322     * @param version
323     * @param concurrencyControl
324     * @param model
325     * @return ModelAndView
326     */
327    @PostMapping(name = "create")
328    public ModelAndView create(@ModelAttribute Category category, @RequestParam(value = "productsIds", required = false) List<Long> products, @RequestParam("parentVersion") Long version, @RequestParam(value = "concurrency", required = false, defaultValue = "") String concurrencyControl, Model model) {
329        // Remove empty values
330        if (products != null) {
331            for (Iterator<Long> iterator = products.iterator(); iterator.hasNext(); ) {
332                if (iterator.next() == null) {
333                    iterator.remove();
334                }
335            }
336        }
337        // Concurrency control
338        if (!Objects.equals(version, category.getVersion()) && StringUtils.isEmpty(concurrencyControl)) {
339            populateForm(model);
340            // Obtain the selected books and include them in the author that will be
341            // included in the view
342            if (products != null) {
343                category.setProducts(new HashSet<Product>(getProductService().findAll(products)));
344            } else {
345                category.setProducts(new HashSet<Product>());
346            }
347            // Reset the version to prevent update
348            category.setVersion(version);
349            // Include the updated element in the model
350            model.addAttribute("category", category);
351            model.addAttribute("concurrency", true);
352            return new ModelAndView("categories/products/create");
353        } else if (!Objects.equals(version, category.getVersion()) && "discard".equals(concurrencyControl)) {
354            populateForm(model);
355            // Provide the original element from the Database
356            model.addAttribute("category", category);
357            model.addAttribute("concurrency", false);
358            return new ModelAndView("categories/products/create");
359        }
360        getCategoryService().setProducts(category, products);
361        return new ModelAndView("redirect:" + getCollectionLink().to(CategoriesCollectionThymeleafLinkFactory.LIST).toUriString());
362    }
