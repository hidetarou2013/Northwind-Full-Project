101    /**
102     * TODO Auto-generated method documentation
103     *
104     * @param globalSearch
105     * @param pageable
106     * @return Page
107     */
108    public Page<Supplier> findAll(GlobalSearch globalSearch, Pageable pageable) {
109        QSupplier supplier = QSupplier.supplier;
110        JPQLQuery<Supplier> query = from(supplier);
111        Path<?>[] paths = new Path<?>[] { supplier.city, supplier.country, supplier.region, supplier.companyName, supplier.contactName, supplier.contactTitle, supplier.address, supplier.postalCode, supplier.phone, supplier.fax, supplier.web };
112        applyGlobalSearch(globalSearch, query, paths);
113        AttributeMappingBuilder mapping = buildMapper().map(CITY, supplier.city).map(COUNTRY, supplier.country).map(REGION, supplier.region).map(COMPANY_NAME, supplier.companyName).map(CONTACT_NAME, supplier.contactName).map(CONTACT_TITLE, supplier.contactTitle).map(ADDRESS, supplier.address).map(POSTAL_CODE, supplier.postalCode).map(PHONE, supplier.phone).map(FAX, supplier.fax).map(WEB, supplier.web);
114        applyPagination(pageable, query, mapping);
115        applyOrderById(query);
116        return loadPage(query, pageable, supplier);
117    }
