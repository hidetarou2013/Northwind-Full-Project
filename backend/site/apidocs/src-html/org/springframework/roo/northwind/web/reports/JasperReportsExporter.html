001package org.springframework.roo.northwind.web.reports;
002
003import java.io.IOException;
004
005import javax.servlet.http.HttpServletResponse;
006
007import net.sf.jasperreports.engine.JRException;
008import net.sf.jasperreports.engine.JasperPrint;
009import net.sf.jasperreports.export.Exporter;
010
011/**
012 * = JasperReportsExporter
013 * 
014 * This interface defines the operations for a JasperReport exporter.
015 * 
016 * JasperReports library already provides an Exporter interface called
017 * {@link Exporter}. However, it doesn't provides an operation that writes the
018 * exported JasperReport into the {@link HttpServletResponse}.
019 */
020public interface JasperReportsExporter {
021
022        /**
023         * This operation must be implemented by every JasperReport exporter to be
024         * able to write a generated report into a the {@link HttpServletResponse}}.
025         * 
026         * @param jp
027         *            The generated JasperReport.
028         * @param fileName
029         *            The fileName of the exported JasperReport
030         * @param response
031         *            The HttpServletResponse where generated report has been
032         *            written
033         * @throws JRException
034         *             during JasperReport export.
035         * @throws IOException
036         *             when writes the ByteArrayOutputStream into the
037         *             HttpServletResponse
038         */
039        public void export(JasperPrint jp, String fileName, HttpServletResponse response) throws JRException, IOException;
040
041}
