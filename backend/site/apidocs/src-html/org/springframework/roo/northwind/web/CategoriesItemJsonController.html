001package org.springframework.roo.northwind.web;

030@RooController(entity = Category.class, pathPrefix = "/api", type = ControllerType.ITEM)
031@RooJSON
032@RestController
033@RequestMapping(value = "/api/categories/{category}", name = "CategoriesItemJsonController", produces = MediaType.APPLICATION_JSON_VALUE)
034public class CategoriesItemJsonController {
035
036    /**
037     * TODO Auto-generated attribute documentation
038     *
039     */
040    private CategoryService categoryService;
041
042    /**
043     * TODO Auto-generated constructor documentation
044     *
045     * @param categoryService
046     */
047    @Autowired
048    public CategoriesItemJsonController(CategoryService categoryService) {
049        this.categoryService = categoryService;
050    }
051
052    /**
053     * TODO Auto-generated method documentation
054     *
055     * @return CategoryService
056     */
057    public CategoryService getCategoryService() {
058        return categoryService;
059    }
060
061    /**
062     * TODO Auto-generated method documentation
063     *
064     * @param categoryService
065     */
066    public void setCategoryService(CategoryService categoryService) {
067        this.categoryService = categoryService;
068    }
069
070    /**
071     * TODO Auto-generated method documentation
072     *
073     * @param id
074     * @return Category
075     */
076    @ModelAttribute
077    public Category getCategory(@PathVariable("category") Long id) {
078        Category category = categoryService.findOne(id);
079        if (category == null) {
080            throw new NotFoundException(String.format("Category with identifier '%s' not found", id));
081        }
082        return category;
083    }
084
085    /**
086     * TODO Auto-generated method documentation
087     *
088     * @param category
089     * @return ResponseEntity
090     */
091    @GetMapping(name = "show")
092    public ResponseEntity<?> show(@ModelAttribute Category category) {
093        return ResponseEntity.ok(category);
094    }
095
096    /**
097     * TODO Auto-generated method documentation
098     *
099     * @param category
100     * @return UriComponents
101     */
102    public static UriComponents showURI(Category category) {
103        return MvcUriComponentsBuilder.fromMethodCall(MvcUriComponentsBuilder.on(CategoriesItemJsonController.class).show(category)).buildAndExpand(category.getId()).encode();
104    }
105
106    /**
107     * TODO Auto-generated method documentation
108     *
109     * @param storedCategory
110     * @param category
111     * @param result
112     * @return ResponseEntity
113     */
114    @PutMapping(name = "update")
115    public ResponseEntity<?> update(@ModelAttribute Category storedCategory, @Valid @RequestBody Category category, BindingResult result) {
116        if (result.hasErrors()) {
117            return ResponseEntity.status(HttpStatus.CONFLICT).body(result);
118        }
119        category.setId(storedCategory.getId());
120        getCategoryService().save(category);
121        return ResponseEntity.ok().build();
122    }
123
124    /**
125     * TODO Auto-generated method documentation
126     *
127     * @param category
128     * @return ResponseEntity
129     */
130    @DeleteMapping(name = "delete")
131    public ResponseEntity<?> delete(@ModelAttribute Category category) {
132        getCategoryService().delete(category);
133        return ResponseEntity.ok().build();
134    }
135}
