230    @GetMapping(produces = Datatables.MEDIA_TYPE, name = "datatables", value = "/dt")
231    @ResponseBody
232    public ResponseEntity<ConvertedDatatablesData<Store>> datatables(DatatablesColumns datatablesColumns, GlobalSearch search, DatatablesPageable pageable, @RequestParam("draw") Integer draw) {
233        Page<Store> stores = getStoreService().findAll(search, pageable);
234        long totalStoresCount = stores.getTotalElements();
235        if (search != null && StringUtils.isNotBlank(search.getText())) {
236            totalStoresCount = getStoreService().count();
237        }
238        ConvertedDatatablesData<Store> datatablesData = new ConvertedDatatablesData<Store>(stores, totalStoresCount, draw, getConversionService(), datatablesColumns);
239        return ResponseEntity.ok(datatablesData);
240    }

357    /**
358     * Method that obtains the filtered and ordered records using the Datatables information and
359     * export them to a new report file. (It ignores the current pagination).
360     *
361     * To generate the report file it uses the `DynamicJasper` library
362     * (http://dynamicjasper.com). This library allows developers to generate reports dynamically
363     * without use an specific template to each entity.
364     *
365     * To customize the appearance of ALL generated reports, you could customize the
366     * "export_default.jrxml" template located in "src/main/resources/templates/reports/". However,
367     * if you want to customize the appearance of this specific report, you could create a new
368     * ".jrxml" file and provide it to the library replacing the `builder.setTemplateFile();`
369     * operation used in this implementation.
370     *
371     * @param search GlobalSearch that contains the filter provided by the Datatables component.
372     * @param pageable Pageable that contains the Sort info provided by the Datatabes component.
373     * @param datatablesColumns Columns displayed in the Datatables component.
374     * @param response The HttpServletResponse.
375     * @param exporter An specific JasperReportsExporter to be used during export process.
376     * @param fileName The final filename to use.
377     * @param locale The current Locale in the view context.
378     */
379    public void export(GlobalSearch search, @PageableDefault(size = 2147483647) Pageable pageable, String[] datatablesColumns, HttpServletResponse response, JasperReportsExporter exporter, String fileName, Locale locale) {
380        // Obtain the filtered and ordered elements
381        Page<Store> stores = getStoreService().findAll(search, pageable);
382        // Prevent generation of reports with empty data
383        if (stores == null || stores.getContent().isEmpty()) {
384            return;
385        }
386        // Creates a new ReportBuilder using DynamicJasper library
387        FastReportBuilder builder = new FastReportBuilder();
388        // IMPORTANT: By default, this application uses "export_default.jrxml"
389        // to generate all reports. If you want to customize this specific report,
390        // create a new ".jrxml" template and customize it. (Take in account the
391        // DynamicJasper restrictions:
392        // http://dynamicjasper.com/2010/10/06/how-to-use-custom-jrxml-templates/)
393        builder.setTemplateFile("templates/reports/export_default.jrxml");
394        // The generated report will display the same columns as the Datatables component.
395        // However, this is not mandatory. You could edit this code if you want to ignore
396        // the provided datatablesColumns
397        if (datatablesColumns != null) {
398            for (String column : datatablesColumns) {
399                // Delegates in addColumnToReportBuilder to include each datatables column
400                // to the report builder
401                addColumnToReportBuilder(column, builder, locale, fileName);
402            }
403        }
404        // This property resizes the columns to use full width page.
405        // Set false value if you want to use the specific width of each column.
406        builder.setUseFullPageWidth(true);
407        // Creates a new Jasper Reports Datasource using the obtained elements
408        JRDataSource ds = new JRBeanCollectionDataSource(stores.getContent());
409        // Generates the JasperReport
410        JasperPrint jp;
411        try {
412            jp = DynamicJasperHelper.generateJasperPrint(builder.build(), new ClassicLayoutManager(), ds);
413        } catch (JRException e) {
414            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
415            throw new ExportingErrorException(errorMessage);
416        }
417        // Converts the JaspertReport element to a ByteArrayOutputStream and
418        // write it into the response stream using the provided JasperReportExporter
419        try {
420            exporter.export(jp, fileName, response);
421        } catch (JRException e) {
422            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
423            throw new ExportingErrorException(errorMessage);
424        } catch (IOException e) {
425            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
426            throw new ExportingErrorException(errorMessage);
427        }
428    }

481    /**
482     * This method contains all the entity fields that are able to be displayed in a
483     * report. The developer could add a new column to the report builder providing the
484     * field name and the builder where the new field will be added as column.
485     *
486     * @param columnName the field name to show as column
487     * @param builder The builder where the new field will be added as column.
488     */
489    public void addColumnToReportBuilder(String columnName, FastReportBuilder builder, Locale locale, String fileName) {
490        try {
491            if (columnName.equals("id")) {
492                builder.addColumn(getMessageSource().getMessage("label_store_id", null, "Id", locale), "id", Long.class.getName(), 50);
493            } else if (columnName.equals("version")) {
494                builder.addColumn(getMessageSource().getMessage("label_store_version", null, "Version", locale), "version", Long.class.getName(), 100);
495            } else if (columnName.equals("name")) {
496                builder.addColumn(getMessageSource().getMessage("label_store_name", null, "Name", locale), "name", String.class.getName(), 100);
497            } else if (columnName.equals("address")) {
498                builder.addColumn(getMessageSource().getMessage("label_store_address", null, "Address", locale), "address", String.class.getName(), 100);
499            } else if (columnName.equals("postalCode")) {
500                builder.addColumn(getMessageSource().getMessage("label_store_postalcode", null, "Postal Code", locale), "postalCode", String.class.getName(), 100);
501            } else if (columnName.equals("phone")) {
502                builder.addColumn(getMessageSource().getMessage("label_store_phone", null, "Phone", locale), "phone", String.class.getName(), 100);
503            }
504        } catch (ColumnBuilderException e) {
505            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
506            throw new ExportingErrorException(errorMessage);
507        } catch (ClassNotFoundException e) {
508            String errorMessage = getMessageSource().getMessage("error_exportingErrorException", new Object[] { StringUtils.substringAfterLast(fileName, ".").toUpperCase() }, String.format("Error while exporting data to StringUtils file", StringUtils.substringAfterLast(fileName, ".").toUpperCase()), locale);
509            throw new ExportingErrorException(errorMessage);
510        }
511    }
