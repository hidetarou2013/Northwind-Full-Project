301    /**
302     * TODO Auto-generated method documentation
303     *
304     * @param city
305     * @param customerOrders
306     * @return City
307     */
308    @Transactional
309    public City setCustomerOrders(City city, Iterable<Long> customerOrders) {
310        List<CustomerOrder> items = getCustomerOrderService().findAll(customerOrders);
311        Set<CustomerOrder> currents = city.getCustomerOrders();
312        Set<CustomerOrder> toRemove = new HashSet<CustomerOrder>();
313        for (Iterator<CustomerOrder> iterator = currents.iterator(); iterator.hasNext(); ) {
314            CustomerOrder nextCustomerOrder = iterator.next();
315            if (items.contains(nextCustomerOrder)) {
316                items.remove(nextCustomerOrder);
317            } else {
318                toRemove.add(nextCustomerOrder);
319            }
320        }
321        city.removeFromCustomerOrders(toRemove);
322        city.addToCustomerOrders(items);
323        // Force the version update of the parent side to know that the parent has changed
324        // because it has new books assigned
325        city.setVersion(city.getVersion() + 1);
326        return getCityRepository().save(city);
327    }

413    /**
414     * TODO Auto-generated method documentation
415     *
416     * @param city
417     */
418    @Transactional
419    public void delete(City city) {
420        // Clear bidirectional many-to-one child relationship with Region
421        if (city.getRegion() != null) {
422            city.getRegion().getCities().remove(city);
423        }
424        // Clear bidirectional one-to-many parent relationship with CustomerOrder
425        for (CustomerOrder item : city.getCustomerOrders()) {
426            item.setCity(null);
427        }
428        // Clear bidirectional one-to-many parent relationship with Party
429        for (Party item : city.getParties()) {
430            item.setCity(null);
431        }
432        // Clear bidirectional one-to-many parent relationship with Store
433        for (Store item : city.getStores()) {
434            item.setCity(null);
435        }
436        // Clear bidirectional one-to-many parent relationship with Supplier
437        for (Supplier item : city.getSuppliers()) {
438            item.setCity(null);
439        }
440        getCityRepository().delete(city);
441    }
